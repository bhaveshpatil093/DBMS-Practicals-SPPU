<body>
    <h2>Student Management System</h2>
    <div id="msg" style="color:green;"></div>
    
    <form id="form">
        ID: <input type="number" id="id"><br><br>
        Name: <input type="text" id="name" required><br><br>
        Email: <input type="email" id="email" required><br><br>
        Course: <input type="text" id="course" required><br><br>
        
        <button type="button" onclick="add()">Add</button>
        <button type="button" onclick="update()">Update</button>
        <button type="button" onclick="del()">Delete</button>
        <button type="button" onclick="clear()">Clear</button>
    </form>
    
    <h3>Student Records</h3>
    <table border="1" cellpadding="10">
        <tr><th>ID</th><th>Name</th><th>Email</th><th>Course</th></tr>
        <tbody id="data"></tbody>
    </table>

    <script>
        let students = JSON.parse(localStorage.getItem('students')) || [];
        let nextId = students.length > 0 ? Math.max(...students.map(s => s.id)) + 1 : 1;

        function show() {
            let html = '';
            students.forEach(s => {
                html += `<tr onclick="select(${s.id})">
                    <td>${s.id}</td><td>${s.name}</td><td>${s.email}</td><td>${s.course}</td>
                </tr>`;
            });
            document.getElementById('data').innerHTML = html || '<tr><td colspan="4">No records</td></tr>';
        }

        function add() {
            let name = document.getElementById('name').value;
            let email = document.getElementById('email').value;
            let course = document.getElementById('course').value;
            
            if (!name || !email || !course) return alert('Fill all fields!');
            if (students.some(s => s.email === email)) return alert('Email exists!');
            
            students.push({id: nextId++, name, email, course});
            save();
            msg('Added successfully');
        }

        function update() {
            let id = parseInt(document.getElementById('id').value);
            let name = document.getElementById('name').value;
            let email = document.getElementById('email').value;
            let course = document.getElementById('course').value;
            
            if (!id) return alert('Enter ID or select from table!');
            if (!name || !email || !course) return alert('Fill all fields!');
            
            let i = students.findIndex(s => s.id === id);
            if (i === -1) return alert('Student not found!');
            
            students[i] = {id, name, email, course};
            save();
            msg('Updated successfully');
        }

        function del() {
            let id = parseInt(document.getElementById('id').value);
            if (!id) return alert('Enter ID or select from table!');
            
            let i = students.findIndex(s => s.id === id);
            if (i === -1) return alert('Student not found!');
            
            if (confirm('Delete this student?')) {
                students.splice(i, 1);
                save();
                msg('Deleted successfully');
            }
        }

        function select(id) {
            let s = students.find(st => st.id === id);
            document.getElementById('id').value = s.id;
            document.getElementById('name').value = s.name;
            document.getElementById('email').value = s.email;
            document.getElementById('course').value = s.course;
        }

        function clear() {
            document.getElementById('form').reset();
            document.getElementById('id').value = '';
        }

        function save() {
            localStorage.setItem('students', JSON.stringify(students));
            show();
            clear();
        }

        function msg(text) {
            document.getElementById('msg').textContent = text;
            setTimeout(() => document.getElementById('msg').textContent = '', 3000);
        }

        show();
    </script>
</body>

Demonstrate database connectivity with a frontend language to perform CRUD operations (Create, Read, Update, Delete).

WHAT IS DATABASE CONNECTIVITY?
Database Connectivity = Communication between Frontend (user interface) and Database (data storage)
User Interface (HTML) ←→ Programming Logic (JavaScript/Python/Java) ←→ Database (MySQL/Oracle)

UPDATE Operation:

2. Modifies data
3. Clicks the "Update" button
4. Finds record by ID
5. Updates the database
6. Refreshes the display

DELETE Operation:
1. User selects student
